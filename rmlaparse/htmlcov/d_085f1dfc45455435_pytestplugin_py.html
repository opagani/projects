96 def pytest_report_header(config): 

97 if config.option.verbose < 1 and settings.default.verbosity < Verbosity.verbose: 

98 return None 

99 profile = config.getoption(LOAD_PROFILE_OPTION) 

100 if not profile: 

101 profile = settings._current_profile 

102 settings_str = settings.get_profile(profile).show_changed() 

103 if settings_str != "": 

104 settings_str = f" -> {settings_str}" 

105 return f"hypothesis profile {profile!r}{settings_str}"

107 def pytest_configure(config): 

108 core.running_under_pytest = True 

109 profile = config.getoption(LOAD_PROFILE_OPTION) 

110 if profile: 

111 settings.load_profile(profile) 

112 verbosity_name = config.getoption(VERBOSITY_OPTION) 

113 if verbosity_name and verbosity_name != settings.default.verbosity.name: 

114 verbosity_value = Verbosity[verbosity_name] 

115 name = f"{settings._current_profile}-with-{verbosity_name}-verbosity" 

116 # register_profile creates a new profile, exactly like the current one, 

117 # with the extra values given (in this case 'verbosity') 

118 settings.register_profile(name, verbosity=verbosity_value) 

119 settings.load_profile(name) 

120 if ( 

121 config.getoption(EXPLAIN_OPTION) 

122 and Phase.explain not in settings.default.phases 

123 ): 

124 name = f"{settings._current_profile}-with-explain-phase" 

125 phases = settings.default.phases + (Phase.explain,) 

126 settings.register_profile(name, phases=phases) 

127 settings.load_profile(name) 

128 

129 seed = config.getoption(SEED_OPTION) 

130 if seed is not None: 

131 try: 

132 seed = int(seed) 

133 except ValueError: 

134 pass 

135 core.global_force_seed = seed 

136 config.addinivalue_line("markers", "hypothesis: Tests which use hypothesis.")

138 @pytest.hookimpl(hookwrapper=True) 

139 def pytest_runtest_call(item): 

140 if not hasattr(item, "obj"): 

141 yield 

142 elif not is_hypothesis_test(item.obj): 

143 # If @given was not applied, check whether other hypothesis 

144 # decorators were applied, and raise an error if they were. 

145 if getattr(item.obj, "is_hypothesis_strategy_function", False): 

146 raise InvalidArgument( 

147 f"{item.nodeid} is a function that returns a Hypothesis strategy, " 

148 "but pytest has collected it as a test function. This is useless " 

149 "as the function body will never be executed. To define a test " 

150 "function, use @given instead of @composite." 

151 ) 

152 message = "Using `@%s` on a test without `@given` is completely pointless." 

153 for name, attribute in [ 

154 ("example", "hypothesis_explicit_examples"), 

155 ("seed", "_hypothesis_internal_use_seed"), 

156 ("settings", "_hypothesis_internal_settings_applied"), 

157 ("reproduce_example", "_hypothesis_internal_use_reproduce_failure"), 

158 ]: 

159 if hasattr(item.obj, attribute): 

160 raise InvalidArgument(message % (name,)) 

161 yield 

162 else: 

163 # Retrieve the settings for this test from the test object, which 

164 # is normally a Hypothesis wrapped_test wrapper. If this doesn't 

165 # work, the test object is probably something weird 

166 # (e.g a stateful test wrapper), so we skip the function-scoped 

167 # fixture check. 

168 settings = getattr(item.obj, "_hypothesis_internal_use_settings", None) 

169 

170 # Check for suspicious use of function-scoped fixtures, but only 

171 # if the corresponding health check is not suppressed. 

172 if ( 

173 settings is not None 

174 and HealthCheck.function_scoped_fixture 

175 not in settings.suppress_health_check 

176 ): 

177 # Warn about function-scoped fixtures, excluding autouse fixtures because 

178 # the advice is probably not actionable and the status quo seems OK... 

179 # See https://github.com/HypothesisWorks/hypothesis/issues/377 for detail. 

180 msg = ( 

181 "%s uses the %r fixture, which is reset between function calls but not " 

182 "between test cases generated by `@given(...)`. You can change it to " 

183 "a module- or session-scoped fixture if it is safe to reuse; if not " 

184 "we recommend using a context manager inside your test function. See " 

185 "https://docs.pytest.org/en/latest/how-to/fixtures.html" 

186 "#scope-sharing-fixtures-across-classes-modules-packages-or-session " 

187 "for details on fixture scope." 

188 ) 

189 argnames = None 

190 for fx_defs in item._request._fixturemanager.getfixtureinfo( 

191 node=item, func=item.function, cls=None 

192 ).name2fixturedefs.values(): 

193 if argnames is None: 

194 argnames = frozenset(signature(item.function).parameters) 

195 for fx in fx_defs: 

196 if fx.argname in argnames: 

197 active_fx = item._request._get_active_fixturedef(fx.argname) 

198 if active_fx.scope == "function": 

199 fail_health_check( 

200 settings, 

201 msg % (item.nodeid, fx.argname), 

202 HealthCheck.function_scoped_fixture, 

203 ) 

204 

205 if item.get_closest_marker("parametrize") is not None: 

206 # Give every parametrized test invocation a unique database key 

207 key = item.nodeid.encode() 

208 item.obj.hypothesis.inner_test._hypothesis_internal_add_digest = key 

209 

210 store = StoringReporter(item.config)

213 stats["nodeid"] = item.nodeid 

214 item.hypothesis_statistics = base64.b64encode(

225 def pytest_runtest_makereport(item, call): 

226 report = (yield).get_result() 

227 if hasattr(item, "hypothesis_report_information"): 

228 report.sections.append(

233 try: 

234 item.config._xml.add_global_property(name, item.hypothesis_statistics)

238 report.user_properties.append((name, item.hypothesis_statistics))

242 if pytest_html is not None: # pragma: no cover

252 def report(properties): 

253 for name, value in properties: 

254 if name.startswith("hypothesis-statistics-"): 

255 if hasattr(value, "uniobj"): 

256 # Under old versions of pytest, `value` was a `py.xml.raw` 

257 # rather than a string, so we get the (unicode) string off it. 

258 value = value.uniobj 

259 line = base64.b64decode(value.encode()).decode() + "\n\n" 

260 terminalreporter.write_line(line) 

261 

262 try: 

263 global_properties = terminalreporter.config._xml.global_properties 

264 except AttributeError: 

265 # terminalreporter.stats is a dict, where the empty string appears to 

266 # always be the key for a list of _pytest.reports.TestReport objects 

267 for test_report in terminalreporter.stats.get("", []): 

268 if test_report.when == "teardown": 

269 report(test_report.user_properties) 

270 else: 

271 report(global_properties)
